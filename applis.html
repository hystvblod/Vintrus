// Moteur i18n simple pour test // Supposons window.i18n déjà chargé par un document.addEventListener("DOMContentLoaded", function() { document.querySelectorAll("[data-i18n]").forEach(el => { const key = el.getAttribute("data-i18n"); if (window.i18n && window.i18n[key]) el.textContent = window.i18n[key]; }); // Et titre dynamique (si jamais) if (window.i18n && window.i18n["apps.title"]) document.title = window.i18n["apps.title"]; });
